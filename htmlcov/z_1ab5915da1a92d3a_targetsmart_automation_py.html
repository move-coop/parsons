1"""**TargetSmart Automation** 

2 

3Parsons provides methods for interacting with TargetSmart Automation Workflows, 

4a solution for executing custom file processing workflows programmatically. In 

5some cases, TargetSmart will provide custom list matching solutions using 

6Automation Workflows. Most TargetSmart clients do not have these workflows and 

7will only use the Developer API. 

8 

9**TargetSmart Developer API versus Automation** 

10 

11TargetSmart's Developer API provides an HTTP-based interface for consuming the 

12general web services that TargetSmart provides. The TargetSmart Automation 

13system solely provides a solution for consuming customized file processing 

14workflows that are provisioned for specific client needs. TargetSmart 

15Automation is based on SFTP instead of HTTP. 

16 

17https://docs.targetsmart.com/my_tsmart/automation/developer.html 

18 

19For most list matching applications, TargetSmart SmartMatch is now the recommended 

20solution. See `TargetSmartAPI.smartmatch`. 

21 

22"""

65 def match( 

66 self, 

67 table, 

68 job_type, 

69 job_name=None, 

70 emails=None, 

71 call_back=None, 

72 remove_files=True, 

73 ): 

74 """Submit a file for custom data processing using the TargetSmart Automation workflow solution. 

75 

76 .. warning:: 

77 Table Columns 

78 

79 Each Automation workflow expects an input file that meets the 

80 layout requirements provided by TargetSmart. The number of columns 

81 and column order is significant. So, if it expected 10 columns and 

82 you only provide 9, it will fail. However, if you provide 10 

83 columns that are out of order, the job may succeed, but with 

84 non-optimal results. You can obtain the layout requirements and 

85 other information about a workflow by visiting the Automation 

86 console in My TargetSmart. Contact `TargetSmart Client Services 

87 <mailto:support@targetsmart.com>`_ for support. 

88 

89 Args: 

90 table: Parsons Table Object 

91 A table object with the required columns. Each workflow type 

92 requires the input file to meet the requirements provided by 

93 TargetSmart. You can locate the input and output layouts for 

94 your available workflows using the My TargetSmart Automation 

95 console. 

96 job_type: str 

97 The workflow name to execute. **This is case sensitive.**. You 

98 can locate the workflow names and other information by visiting 

99 the Automation console in My TargetSmart. 

100 job_name: str 

101 Optional job execution name. 

102 emails: list 

103 A list of emails that will received status notifications. This 

104 is useful in debugging failed jobs. 

105 call_back: str 

106 A callback url to which the status will be posted. See 

107 `TargetSmart documentation <https://docs.targetsmart.com/my_tsmart/automation/developer.html>`_ 

108 for more details. 

109 remove_files: boolean 

110 Remove the configuration, file to be matched and matched file from 

111 the TargetSmart SFTP upon completion or failure of match. 

112 

113 """ 

114 

115 # Generate a match job 

116 job_name = job_name or str(uuid.uuid1()) 

117 

118 try: 

119 # Upload table 

120 self.sftp.put_file(table.to_csv(), f"{self.sftp_dir}/{job_name}_input.csv") 

121 logger.info(f"Table with {table.num_rows} rows uploaded to TargetSmart.") 

122 

123 # Create/upload XML configuration 

124 xml = self.create_job_xml( 

125 job_type, 

126 job_name, 

127 emails=emails, 

128 status_key=job_name, 

129 call_back=call_back, 

130 ) 

131 self.sftp.put_file(xml, f"{self.sftp_dir}/{job_name}.job.xml") 

132 logger.info( 

133 f"Payload uploaded to TargetSmart. Job type: {job_type}. Job name: {job_name}" 

134 ) 

135 

136 # Check xml configuration status 

137 self.poll_config_status(job_name) 

138 

139 # Check the status of the match 

140 self.match_status(job_name) 

141 

142 # Download the resulting file 

143 tbl = Table.from_csv(self.sftp.get_file(f"{self.sftp_dir}/{job_name}_output.csv")) 

144 

145 finally: 

146 # Clean up files 

147 if remove_files: 

148 self.remove_files(job_name) 

149 

150 # Return file as a Table 

151 return tbl

153 def execute(self, *args, **kwargs): 

154 """Most Automation workflows perform list matching. However, it is possible that 

155 a custom workflow might be provisioned for a client for other types of 

156 file processing. The ``execute`` method is provided as an alias for the 

157 ``match`` method which may be a confusing name in these cases. 

158 """ 

159 self.match(*args, **kwargs)

161 def create_job_xml(self, job_type, job_name, emails=None, status_key=None, call_back=None): 

162 # Internal method to create a valid job xml 

163 

164 job = ET.Element("job") 

165 

166 # Generate Base XML 

167 input_file = ET.SubElement(job, "inputfile") 

168 input_file.text = job_name + "_input.csv" 

169 output_file = ET.SubElement(job, "outputfile") 

170 output_file.text = job_name + "_output.csv" 

171 jobtype = ET.SubElement(job, "jobtype", text=job_type) 

172 jobtype.text = job_type 

173 

174 # Add status key 

175 args = ET.SubElement(job, "args") 

176 statuskey = ET.SubElement(args, "arg", name="__status_key") 

177 statuskey.text = status_key or job_name 

178 

179 # Option args 

180 if call_back: 

181 callback = ET.SubElement(args, "arg", name="__http_callback") 

182 callback.text = call_back 

183 

184 if emails: 

185 emails_el = ET.SubElement(args, "arg", name="__emails") 

186 emails_el.text = ",".join(emails) 

187 

188 # Write xml to file object 

189 local_path = create_temp_file(suffix=".xml") 

190 tree = ET.ElementTree(job) 

191 tree.write(local_path) 

192 return local_path

194 def poll_config_status(self, job_name, polling_interval=20): 

195 # Poll the configuration status 

196 

197 while True: 

198 time.sleep(polling_interval) 

199 if self.config_status(job_name): 

200 return True 

201 logger.info(f"Waiting on {job_name} job configuration...")

203 def config_status(self, job_name):

205 # the files in the SFTP directory. 

206 

207 for f in self.sftp.list_directory(remote_path=self.sftp_dir): 

208 if f == f"{job_name}.job.xml.good": 

209 logger.info(f"Match job {job_name} configured.") 

210 return True 

211 

212 elif f == f"{job_name}.job.xml.bad": 

213 logger.info(f"Match job {job_name} configuration error.") 

214 # To Do: Lift up the configuration error. 

215 raise ValueError( 

216 "Job configuration failed. If you provided an email address, you will be sent more details." 

217 ) 

218 

219 else: 

220 pass 

221 

222 return False

224 def match_status(self, job_name, polling_interval=60): 

225 # You could also poll their API for the status, which was what the original 

226 # version of the automation matching did. Note: The polling API is public 

227 # and does expose some metadata. This happens regardless of anything that 

228 # we do. However, the actually data is only exposed on the secure SFTP. 

229 

230 while True: 

231 logger.debug("Match running...") 

232 for file_name in self.sftp.list_directory(remote_path=self.sftp_dir):

234 xml_file = self.sftp.get_file(f"{self.sftp_dir}/{job_name}.finish.xml") 

235 with Path(xml_file).open(mode="rb") as x: 

236 xml = xmltodict.parse(x, dict_constructor=dict) 

237 

238 if xml["jobcontext"]["state"] == "error": 

239 # To Do: Parse these in a pretty way 

240 logger.info(f"Match Error: {xml['jobcontext']['errors']}") 

241 raise ValueError(f"Match job failed. {xml['jobcontext']['errors']}") 

242 

243 elif xml["jobcontext"]["state"] == "success": 

244 logger.info("Match complete.") 

245 

246 return True

250 def remove_files(self, job_name): 

251 # Remove all of the files for the match. 

252 

253 for file_name in self.sftp.list_directory(remote_path=self.sftp_dir): 

254 if job_name in file_name: 

255 self.sftp.remove_file(f"{self.sftp_dir}/{file_name}") 

256 logger.info(f"{file_name} removed from SFTP.")
