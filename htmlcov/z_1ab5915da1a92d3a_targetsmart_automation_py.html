1"""**TargetSmart Automation** 

2 

3Parsons provides methods for interacting with TargetSmart Automation Workflows, 

4a solution for executing custom file processing workflows programmatically. In 

5some cases, TargetSmart will provide custom list matching solutions using 

6Automation Workflows. Most TargetSmart clients do not have these workflows and 

7will only use the Developer API. 

8 

9**TargetSmart Developer API versus Automation** 

10 

11TargetSmart's Developer API provides an HTTP-based interface for consuming the 

12general web services that TargetSmart provides. The TargetSmart Automation 

13system solely provides a solution for consuming customized file processing 

14workflows that are provisioned for specific client needs. TargetSmart 

15Automation is based on SFTP instead of HTTP. 

16 

17https://docs.targetsmart.com/my_tsmart/automation/developer.html 

18 

19For most list matching applications, TargetSmart SmartMatch is now the recommended 

20solution. See `TargetSmartAPI.smartmatch`. 

21 

22"""

64 def match( 

65 self, 

66 table, 

67 job_type, 

68 job_name=None, 

69 emails=None, 

70 call_back=None, 

71 remove_files=True, 

72 ): 

73 """Submit a file for custom data processing using the TargetSmart Automation workflow solution. 

74 

75 .. warning:: 

76 Table Columns 

77 

78 Each Automation workflow expects an input file that meets the 

79 layout requirements provided by TargetSmart. The number of columns 

80 and column order is significant. So, if it expected 10 columns and 

81 you only provide 9, it will fail. However, if you provide 10 

82 columns that are out of order, the job may succeed, but with 

83 non-optimal results. You can obtain the layout requirements and 

84 other information about a workflow by visiting the Automation 

85 console in My TargetSmart. Contact `TargetSmart Client Services 

86 <mailto:support@targetsmart.com>`_ for support. 

87 

88 Args: 

89 table: Parsons Table Object 

90 A table object with the required columns. Each workflow type 

91 requires the input file to meet the requirements provided by 

92 TargetSmart. You can locate the input and output layouts for 

93 your available workflows using the My TargetSmart Automation 

94 console. 

95 job_type: str 

96 The workflow name to execute. **This is case sensitive.**. You 

97 can locate the workflow names and other information by visiting 

98 the Automation console in My TargetSmart. 

99 job_name: str 

100 Optional job execution name. 

101 emails: list 

102 A list of emails that will received status notifications. This 

103 is useful in debugging failed jobs. 

104 call_back: str 

105 A callback url to which the status will be posted. See 

106 `TargetSmart documentation <https://docs.targetsmart.com/my_tsmart/automation/developer.html>`_ 

107 for more details. 

108 remove_files: boolean 

109 Remove the configuration, file to be matched and matched file from 

110 the TargetSmart SFTP upon completion or failure of match. 

111 

112 """ 

113 

114 # Generate a match job 

115 job_name = job_name or str(uuid.uuid1()) 

116 

117 try: 

118 # Upload table 

119 self.sftp.put_file(table.to_csv(), f"{self.sftp_dir}/{job_name}_input.csv") 

120 logger.info(f"Table with {table.num_rows} rows uploaded to TargetSmart.") 

121 

122 # Create/upload XML configuration 

123 xml = self.create_job_xml( 

124 job_type, 

125 job_name, 

126 emails=emails, 

127 status_key=job_name, 

128 call_back=call_back, 

129 ) 

130 self.sftp.put_file(xml, f"{self.sftp_dir}/{job_name}.job.xml") 

131 logger.info( 

132 f"Payload uploaded to TargetSmart. Job type: {job_type}. Job name: {job_name}" 

133 ) 

134 

135 # Check xml configuration status 

136 self.poll_config_status(job_name) 

137 

138 # Check the status of the match 

139 self.match_status(job_name) 

140 

141 # Download the resulting file 

142 tbl = Table.from_csv(self.sftp.get_file(f"{self.sftp_dir}/{job_name}_output.csv")) 

143 

144 finally: 

145 # Clean up files 

146 if remove_files: 

147 self.remove_files(job_name) 

148 

149 # Return file as a Table 

150 return tbl

152 def execute(self, *args, **kwargs): 

153 """Most Automation workflows perform list matching. However, it is possible that 

154 a custom workflow might be provisioned for a client for other types of 

155 file processing. The ``execute`` method is provided as an alias for the 

156 ``match`` method which may be a confusing name in these cases. 

157 """ 

158 self.match(*args, **kwargs)

160 def create_job_xml(self, job_type, job_name, emails=None, status_key=None, call_back=None): 

161 # Internal method to create a valid job xml 

162 

163 job = ET.Element("job") 

164 

165 # Generate Base XML 

166 input_file = ET.SubElement(job, "inputfile") 

167 input_file.text = job_name + "_input.csv" 

168 output_file = ET.SubElement(job, "outputfile") 

169 output_file.text = job_name + "_output.csv" 

170 jobtype = ET.SubElement(job, "jobtype", text=job_type) 

171 jobtype.text = job_type 

172 

173 # Add status key 

174 args = ET.SubElement(job, "args") 

175 statuskey = ET.SubElement(args, "arg", name="__status_key") 

176 statuskey.text = status_key or job_name 

177 

178 # Option args 

179 if call_back: 

180 callback = ET.SubElement(args, "arg", name="__http_callback") 

181 callback.text = call_back 

182 

183 if emails: 

184 emails_el = ET.SubElement(args, "arg", name="__emails") 

185 emails_el.text = ",".join(emails) 

186 

187 # Write xml to file object 

188 local_path = create_temp_file(suffix=".xml") 

189 tree = ET.ElementTree(job) 

190 tree.write(local_path) 

191 return local_path

193 def poll_config_status(self, job_name, polling_interval=20): 

194 # Poll the configuration status 

195 

196 while True: 

197 time.sleep(polling_interval) 

198 if self.config_status(job_name): 

199 return True 

200 logger.info(f"Waiting on {job_name} job configuration...")

202 def config_status(self, job_name):

205 

206 for f in self.sftp.list_directory(remote_path=self.sftp_dir): 

207 if f == f"{job_name}.job.xml.good": 

208 logger.info(f"Match job {job_name} configured.") 

209 return True 

210 

211 elif f == f"{job_name}.job.xml.bad": 

212 logger.info(f"Match job {job_name} configuration error.") 

213 # To Do: Lift up the configuration error. 

214 raise ValueError( 

215 "Job configuration failed. If you provided an email" 

216 "address, you will be sent more details." 

217 ) 

218 

219 else: 

220 pass 

221 

222 return False

224 def match_status(self, job_name, polling_interval=60): 

225 # You could also poll their API for the status, which was what the original 

226 # version of the automation matching did. Note: The polling API is public

228 # we do. However, the actually data is only exposed on the secure SFTP.

231 logger.debug("Match running...") 

232 for file_name in self.sftp.list_directory(remote_path=self.sftp_dir):

234 xml_file = self.sftp.get_file(f"{self.sftp_dir}/{job_name}.finish.xml") 

235 with open(xml_file, "rb") as x: 

236 xml = xmltodict.parse(x, dict_constructor=dict) 

237 

238 if xml["jobcontext"]["state"] == "error":

240 logger.info(f"Match Error: {xml['jobcontext']['errors']}") 

241 raise ValueError(f"Match job failed. {xml['jobcontext']['errors']}") 

242 

243 elif xml["jobcontext"]["state"] == "success": 

244 logger.info("Match complete.")

250 def remove_files(self, job_name): 

251 # Remove all of the files for the match. 

252 

253 for file_name in self.sftp.list_directory(remote_path=self.sftp_dir): 

254 if job_name in file_name: 

255 self.sftp.remove_file(f"{self.sftp_dir}/{file_name}") 

256 logger.info(f"{file_name} removed from SFTP.")
