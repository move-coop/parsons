25 """ 

26 Generate a Postgres connection. 

27 The connection is set up as a python "context manager", so it will be closed 

28 automatically (and all queries committed) when the connection goes out of scope. 

29 

30 When using the connection, make sure to put it in a ``with`` block (necessary for 

31 any context manager): 

32 ``with pg.connection() as conn:`` 

33 

34 `Returns:` 

35 Psycopg2 `connection` object 

36 """

39 conn = psycopg2.connect(

42 host=self.host,

44 port=self.port,

67 def query(self, sql: str, parameters: Optional[list] = None) -> Optional[Table]: 

68 """ 

69 Execute a query against the database. Will return ``None`` if the query returns zero rows. 

70 

71 To include python variables in your query, it is recommended to pass them as parameters, 

72 following the `psycopg style <http://initd.org/psycopg/docs/usage.html#passing-parameters-to-sql-queries>`_. 

73 Using the ``parameters`` argument ensures that values are escaped properly, and avoids SQL 

74 injection attacks. 

75 

76 **Parameter Examples** 

77 

78 .. code-block:: python 

79 

80 # Note that the name contains a quote, which could break your query if not escaped 

81 # properly. 

82 name = "Beatrice O'Brady" 

83 sql = "SELECT * FROM my_table WHERE name = %s" 

84 rs.query(sql, parameters=[name]) 

85 

86 .. code-block:: python 

87 

88 names = ["Allen Smith", "Beatrice O'Brady", "Cathy Thompson"] 

89 placeholders = ', '.join('%s' for item in names) 

90 sql = f"SELECT * FROM my_table WHERE name IN ({placeholders})" 

91 rs.query(sql, parameters=names) 

92 

93 `Args:` 

94 sql: str 

95 A valid SQL statement 

96 parameters: list 

97 A list of python variables to be converted into SQL values in your query 

98 

99 `Returns:` 

100 Parsons Table 

101 See :ref:`parsons-table` for output options. 

102 

103 """ 

104 

105 with self.connection() as connection: 

106 return self.query_with_connection(sql, connection, parameters=parameters)

108 def query_with_connection(self, sql, connection, parameters=None, commit=True): 

109 """ 

110 Execute a query against the database, with an existing connection. Useful for batching 

111 queries together. Will return ``None`` if the query returns zero rows. 

112 

113 `Args:` 

114 sql: str 

115 A valid SQL statement 

116 connection: obj 

117 A connection object obtained from ``redshift.connection()`` 

118 parameters: list 

119 A list of python variables to be converted into SQL values in your query 

120 commit: boolean 

121 Whether to commit the transaction immediately. If ``False`` the transaction will 

122 be committed when the connection goes out of scope and is closed (or you can 

123 commit manually with ``connection.commit()``). 

124 

125 `Returns:` 

126 Parsons Table 

127 See :ref:`parsons-table` for output options. 

128 """ 

129 

130 with self.cursor(connection) as cursor: 

131 logger.debug(f"SQL Query: {sql}") 

132 cursor.execute(sql, parameters) 

133 

134 if commit: 

135 connection.commit() 

136 

137 # If the cursor is empty, don't cause an error 

138 if not cursor.description: 

139 logger.debug("Query returned 0 rows") 

140 return None 

141 

142 else: 

143 # Fetch the data in batches, and "pickle" the rows to a temp file. 

144 # (We pickle rather than writing to, say, a CSV, so that we maintain 

145 # all the type information for each field.) 

146 

147 temp_file = files.create_temp_file() 

148 

149 with open(temp_file, "wb") as f: 

150 # Grab the header 

151 header = [i[0] for i in cursor.description] 

152 pickle.dump(header, f) 

153 

154 while True: 

155 batch = cursor.fetchmany(QUERY_BATCH_SIZE) 

156 if not batch: 

157 break 

158 

159 logger.debug(f"Fetched {len(batch)} rows.") 

160 for row in batch: 

161 pickle.dump(list(row), f) 

162 

163 # Load a Table from the file 

164 final_tbl = Table(petl.frompickle(temp_file)) 

165 

166 logger.debug(f"Query returned {final_tbl.num_rows} rows.") 

167 return final_tbl

169 def _create_table_precheck(self, connection, table_name, if_exists): 

170 """ 

171 Helper to determine what to do when you need a table that may already exist. 

172 

173 `Args:` 

174 connection: obj 

175 A connection object obtained from ``redshift.connection()`` 

176 table_name: str 

177 The table to check 

178 if_exists: str 

179 If the table already exists, either ``fail``, ``append``, ``drop``, 

180 or ``truncate`` the table. 

181 `Returns:` 

182 bool 

183 True if the table needs to be created, False otherwise. 

184 """ 

185 

186 if if_exists not in ["fail", "truncate", "append", "drop"]: 

187 raise ValueError("Invalid value for `if_exists` argument") 

188 

189 # If the table exists, evaluate the if_exists argument for next steps. 

190 if self.table_exists_with_connection(table_name, connection): 

191 if if_exists == "fail": 

192 raise ValueError("Table already exists.") 

193 

194 if if_exists == "truncate": 

195 truncate_sql = f"TRUNCATE TABLE {table_name};" 

196 logger.info(f"Truncating {table_name}.") 

197 self.query_with_connection(truncate_sql, connection, commit=False) 

198 

199 if if_exists == "drop": 

200 logger.info(f"Dropping {table_name}.") 

201 drop_sql = f"DROP TABLE {table_name};" 

202 self.query_with_connection(drop_sql, connection, commit=False) 

203 return True 

204 

205 return False 

206 

207 else: 

208 return True

227 def table_exists_with_connection(self, table_name, connection, view=True): 

228 # Extract the table and schema from this. If no schema is detected then 

229 # will default to the public schema. 

230 try: 

231 schema, table = table_name.split(".", 1) 

232 except ValueError: 

233 schema, table = "public", table_name 

234 

235 with self.cursor(connection) as cursor: 

236 # Check in pg tables for the table 

237 sql = f"""select count(*) from pg_tables where schemaname='{schema}' and 

238 tablename='{table}';""" 

239 

240 cursor.execute(sql) 

241 result = cursor.fetchone()[0] 

242 

243 # Check in the pg_views if it is a view 

244 if view: 

245 sql = f"""select count(*) from pg_views where schemaname='{schema}' and 

246 viewname='{table}';""" 

247 cursor.execute(sql) 

248 result += cursor.fetchone()[0] 

249 

250 # If in either, return boolean 

251 if result >= 1: 

252 return True 

253 else: 

254 return False
