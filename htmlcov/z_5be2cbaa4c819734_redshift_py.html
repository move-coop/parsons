158 def query(self, sql: str, parameters: Optional[list] = None) -> Optional[Table]: 

159 """ 

160 Execute a query against the Redshift database. Will return ``None`` 

161 if the query returns zero rows. 

162 

163 To include python variables in your query, it is recommended to pass them as parameters, 

164 following the `psycopg style <http://initd.org/psycopg/docs/usage.html#passing-parameters-to-sql-queries>`_. 

165 Using the ``parameters`` argument ensures that values are escaped properly, and avoids SQL 

166 injection attacks. 

167 

168 **Parameter Examples** 

169 

170 .. code-block:: python 

171 

172 # Note that the name contains a quote, which could break your query if not escaped 

173 # properly. 

174 name = "Beatrice O'Brady" 

175 sql = "SELECT * FROM my_table WHERE name = %s" 

176 rs.query(sql, parameters=[name]) 

177 

178 .. code-block:: python 

179 

180 names = ["Allen Smith", "Beatrice O'Brady", "Cathy Thompson"] 

181 placeholders = ', '.join('%s' for item in names) 

182 sql = f"SELECT * FROM my_table WHERE name IN ({placeholders})" 

183 rs.query(sql, parameters=names) 

184 

185 `Args:` 

186 sql: str 

187 A valid SQL statement 

188 parameters: list 

189 A list of python variables to be converted into SQL values in your query 

190 

191 `Returns:` 

192 Parsons Table 

193 See :ref:`parsons-table` for output options. 

194 

195 """ 

196 

197 with self.connection() as connection: 

198 return self.query_with_connection(sql, connection, parameters=parameters)

200 def query_with_connection(self, sql, connection, parameters=None, commit=True): 

201 """ 

202 Execute a query against the Redshift database, with an existing connection. 

203 Useful for batching queries together. Will return ``None`` if the query 

204 returns zero rows. 

205 

206 `Args:` 

207 sql: str 

208 A valid SQL statement 

209 connection: obj 

210 A connection object obtained from ``redshift.connection()`` 

211 parameters: list 

212 A list of python variables to be converted into SQL values in your query 

213 commit: boolean 

214 Whether to commit the transaction immediately. If ``False`` the transaction will 

215 be committed when the connection goes out of scope and is closed (or you can 

216 commit manually with ``connection.commit()``). 

217 

218 `Returns:` 

219 Parsons Table 

220 See :ref:`parsons-table` for output options. 

221 """ 

222 

223 # To Do: Have it return an ordered dict to return the 

224 # rows in the correct order 

225 

226 with self.cursor(connection) as cursor: 

227 if "credentials" not in sql: 

228 logger.debug(f"SQL Query: {sql}") 

229 cursor.execute(sql, parameters) 

230 

231 if commit: 

232 connection.commit() 

233 

234 # If the cursor is empty, don't cause an error 

235 if not cursor.description: 

236 logger.debug("Query returned 0 rows") 

237 return None 

238 

239 else: 

240 # Fetch the data in batches, and "pickle" the rows to a temp file. 

241 # (We pickle rather than writing to, say, a CSV, so that we maintain 

242 # all the type information for each field.) 

243 

244 temp_file = files.create_temp_file() 

245 

246 with open(temp_file, "wb") as f: 

247 # Grab the header 

248 header = [i[0] for i in cursor.description] 

249 pickle.dump(header, f) 

250 

251 while True: 

252 batch = cursor.fetchmany(QUERY_BATCH_SIZE) 

253 if not batch: 

254 break 

255 

256 logger.debug(f"Fetched {len(batch)} rows.") 

257 for row in batch: 

258 pickle.dump(list(row), f) 

259 

260 # Load a Table from the file 

261 final_tbl = Table(petl.frompickle(temp_file)) 

262 

263 logger.debug(f"Query returned {final_tbl.num_rows} rows.") 

264 return final_tbl

636 tbl, 

637 table_name, 

638 padding=padding, 

639 distkey=distkey, 

640 sortkey=sortkey, 

641 varchar_max=varchar_max, 

642 columntypes=columntypes, 

643 strict_length=strict_length, 

644 ) 

645 self.query_with_connection(sql, connection, commit=False) 

646 logger.info(f"{table_name} created.") 

647 

648 # If alter_table is True, then alter table if the table column widths 

649 # are wider than the existing table. 

650 if alter_table: 

651 self.alter_varchar_column_widths( 

652 tbl, table_name, drop_dependencies=alter_table_cascade 

653 ) 

654 

655 # Upload the table to S3 

656 key = self.temp_s3_copy( 

657 tbl, 

658 aws_access_key_id=aws_access_key_id, 

659 aws_secret_access_key=aws_secret_access_key, 

660 csv_encoding=csv_encoding, 

661 ) 

662 

663 try: 

664 # Copy to Redshift database. 

665 copy_args = { 

666 "max_errors": max_errors, 

667 "ignoreheader": 1, 

668 "statupdate": statupdate, 

669 "compupdate": compupdate, 

670 "acceptanydate": acceptanydate, 

671 "dateformat": dateformat, 

672 "timeformat": timeformat, 

673 "blanksasnull": blanksasnull, 

674 "nullas": nullas, 

675 "emptyasnull": emptyasnull, 

676 "acceptinvchars": acceptinvchars, 

677 "truncatecolumns": truncatecolumns, 

678 "specifycols": cols, 

679 "aws_access_key_id": aws_access_key_id, 

680 "aws_secret_access_key": aws_secret_access_key, 

681 "compression": "gzip", 

682 "bucket_region": temp_bucket_region, 

683 } 

684 

685 # Copy from S3 to Redshift 

686 sql = self.copy_statement(table_name, self.s3_temp_bucket, key, **copy_args) 

687 sql_censored = sql_helpers.redact_credentials(sql) 

688 

689 logger.debug(f"Copy SQL command: {sql_censored}") 

690 self.query_with_connection(sql, connection, commit=False) 

691 

692 logger.info(f"Data copied to {table_name}.") 

693 

694 # Clean up the S3 bucket. 

695 finally: 

696 if key and cleanup_s3_file: 

697 self.temp_s3_delete(key)

699 def unload( 

700 self, 

701 sql, 

702 bucket, 

703 key_prefix, 

704 manifest=True, 

705 header=True, 

706 delimiter="|", 

707 compression="gzip", 

708 add_quotes=True, 

709 null_as=None, 

710 escape=True, 

711 allow_overwrite=True, 

712 parallel=True, 

713 max_file_size="6.2 GB", 

714 extension=None, 

715 aws_region=None, 

716 format=None, 

717 aws_access_key_id=None, 

718 aws_secret_access_key=None, 

719 ): 

720 r""" 

721 Unload Redshift data to S3 Bucket. This is a more efficient method than running a query 

722 to export data as it can export in parallel and directly into an S3 bucket. Consider 

723 using this for exports of 10MM or more rows. 

724 

725 sql: str 

726 The SQL string to execute to generate the data to unload. 

727 bucket: str 

728 The destination S3 bucket 

729 key_prefix: str 

730 The prefix of the key names that will be written 

731 manifest: boolean 

732 Creates a manifest file that explicitly lists details for the data files 

733 that are created by the UNLOAD process. 

734 header: boolean 

735 Adds a header line containing column names at the top of each output file. 

736 delimiter: str 

737 Specificies the character used to separate fields. Defaults to '|'. 

738 compression: str 

739 One of ``gzip``, ``bzip2`` or ``None``. Unloads data to one or more compressed 

740 files per slice. Each resulting file is appended with a ``.gz`` or ``.bz2`` extension. 

741 add_quotes: boolean 

742 Places quotation marks around each unloaded data field, so that Amazon Redshift 

743 can unload data values that contain the delimiter itself. 

744 null_as: str 

745 Specifies a string that represents a null value in unload files. If this option is 

746 not specified, null values are unloaded as zero-length strings for delimited output. 

747 escape: boolean 

748 For CHAR and VARCHAR columns in delimited unload files, an escape character (\) is 

749 placed before every linefeed, carriage return, escape characters and delimiters. 

750 allow_overwrite: boolean 

751 If ``True``, will overwrite existing files, including the manifest file. If ``False`` 

752 will fail. 

753 parallel: boolean 

754 By default, UNLOAD writes data in parallel to multiple files, according to the number 

755 of slices in the cluster. The default option is ON or TRUE. If PARALLEL is OFF or 

756 FALSE, UNLOAD writes to one or more data files serially, sorted absolutely according 

757 to the ORDER BY clause, if one is used. 

758 max_file_size: str 

759 The maximum size of files UNLOAD creates in Amazon S3. Specify a decimal value between 

760 5 MB and 6.2 GB. 

761 extension: str 

762 This extension will be added to the end of file names loaded to S3 

763 region: str 

764 The AWS Region where the target Amazon S3 bucket is located. REGION is required for 

765 UNLOAD to an Amazon S3 bucket that is not in the same AWS Region as the Amazon Redshift 

766 cluster. 

767 format: str 

768 The format of the unload file (CSV, PARQUET, JSON) - Optional. 

769 aws_access_key_id: 

770 An AWS access key granted to the bucket where the file is located. Not required 

771 if keys are stored as environmental variables. 

772 aws_secret_access_key: 

773 An AWS secret access key granted to the bucket where the file is located. Not 

774 required if keys are stored as environmental variables. 

775 """ 

776 

777 # The sql query is provided between single quotes, therefore single 

778 # quotes within the actual query must be escaped. 

779 # https://docs.aws.amazon.com/redshift/latest/dg/r_UNLOAD.html#unload-parameters 

780 sql = sql.replace("'", "''") 

781 

782 statement = f""" 

783 UNLOAD ('{sql}') to 's3://{bucket}/{key_prefix}' \n 

784 {self.get_creds(aws_access_key_id, aws_secret_access_key)} \n 

785 PARALLEL {parallel} \n 

786 MAXFILESIZE {max_file_size} 

787 """ 

788 statement += "ALLOWOVERWRITE \n" if allow_overwrite else "" 

789 statement += f"REGION {aws_region} \n" if aws_region else "" 

790 statement += "MANIFEST \n" if manifest else "" 

791 statement += f"EXTENSION '{extension}' \n" if extension else "" 

792 

793 # Format-specific parameters 

794 if format: 

795 format = format.lower() 

796 if format == "csv": 

797 statement += f"DELIMITER AS '{delimiter}' \n" if delimiter else "" 

798 statement += f"NULL AS '{null_as}' \n" if null_as else "" 

799 statement += "HEADER \n" if header else "" 

800 statement += "ESCAPE \n" if escape else "" 

801 statement += "FORMAT AS CSV \n" 

802 statement += f"{compression.upper()} \n" if compression else "" 

803 elif format == "parquet": 

804 statement += "FORMAT AS PARQUET \n" 

805 elif format == "json": 

806 statement += "FORMAT AS JSON \n" 

807 statement += f"{compression.upper()} \n" if compression else "" 

808 else: 

809 # Default text file settings 

810 statement += f"DELIMITER AS '{delimiter}' \n" if delimiter else "" 

811 statement += "ADDQUOTES \n" if add_quotes else "" 

812 statement += f"NULL AS '{null_as}' \n" if null_as else "" 

813 statement += "ESCAPE \n" if escape else "" 

814 statement += "HEADER \n" if header else "" 

815 statement += f"{compression.upper()} \n" if compression else "" 

816 

817 logger.info(f"Unloading data to s3://{bucket}/{key_prefix}") 

818 # Censor sensitive data 

819 statement_censored = sql_helpers.redact_credentials(statement) 

820 logger.debug(statement_censored) 

821 

822 return self.query(statement)

883 def generate_manifest( 

884 self, 

885 buckets, 

886 aws_access_key_id=None, 

887 aws_secret_access_key=None, 

888 mandatory=True, 

889 prefix=None, 

890 manifest_bucket=None, 

891 manifest_key=None, 

892 path=None, 

893 ): 

894 """ 

895 Given a list of S3 buckets, generate a manifest file (JSON format). A manifest file 

896 allows you to copy multiple files into a single table at once. Once the manifest is 

897 generated, you can pass it with the :func:`~parsons.redshift.Redshift.copy_s3` method. 

898 

899 AWS keys are not required if ``AWS_ACCESS_KEY_ID`` and 

900 ``AWS_SECRET_ACCESS_KEY`` environmental variables set. 

901 

902 `Args:` 

903 

904 buckets: list or str 

905 A list of buckets or single bucket from which to generate manifest 

906 aws_access_key_id: str 

907 AWS access key id to access S3 bucket 

908 aws_secret_access_key: str 

909 AWS secret access key to access S3 bucket 

910 mandatory: boolean 

911 The mandatory flag indicates whether the Redshift COPY should 

912 terminate if the file does not exist. 

913 prefix: str 

914 Optional filter for key prefixes 

915 manifest_bucket: str 

916 Optional bucket to write manifest file. 

917 manifest_key: str 

918 Optional key name for S3 bucket to write file 

919 

920 `Returns:` 

921 ``dict`` of manifest 

922 """ 

923 

924 from parsons.aws import S3 

925 

926 s3 = S3( 

927 aws_access_key_id=aws_access_key_id, 

928 aws_secret_access_key=aws_secret_access_key, 

929 use_env_token=self.use_env_token, 

930 ) 

931 

932 # Deal with a single bucket being passed, rather than list. 

933 if isinstance(buckets, str): 

934 buckets = [buckets] 

935

938 for bucket in buckets:

940 key_list = s3.list_keys(bucket, prefix=prefix) 

941 for key in key_list:

944 ) 

945 

946 logger.info("Manifest generated.")

950 # Dump the manifest to a temp JSON file 

951 manifest_path = files.create_temp_file() 

952 with open(manifest_path, "w") as manifest_file_obj: 

953 json.dump(manifest, manifest_file_obj, sort_keys=True, indent=4) 

954 

955 # Upload the file to S3 

956 s3.put_file(manifest_bucket, manifest_key, manifest_path) 

957 

958 logger.info(f"Manifest saved to s3://{manifest_bucket}/{manifest_key}") 

959 

960 return manifest

962 def upsert( 

963 self, 

964 table_obj, 

965 target_table, 

966 primary_key, 

967 vacuum=True, 

968 distinct_check=True, 

969 cleanup_temp_table=True, 

970 alter_table=True, 

971 alter_table_cascade=False, 

972 from_s3=False, 

973 distkey=None, 

974 sortkey=None, 

975 **copy_args, 

976 ): 

977 r""" 

978 Preform an upsert on an existing table. An upsert is a function in which rows 

979 in a table are updated and inserted at the same time. 

980 

981 `Args:` 

982 table_obj: obj 

983 A Parsons table object 

984 target_table: str 

985 The schema and table name to upsert 

986 primary_key: str or list 

987 The primary key column(s) of the target table 

988 vacuum: boolean 

989 Re-sorts rows and reclaims space in the specified table. You must be a table owner 

990 or super user to effectively vacuum a table, however the method will not fail 

991 if you lack these priviledges. 

992 distinct_check: boolean 

993 Check if the primary key column is distinct. Raise error if not. 

994 cleanup_temp_table: boolean 

995 A temp table is dropped by default on cleanup. You can set to False for debugging. 

996 alter_table: boolean 

997 Set to False to avoid automatic varchar column resizing to accomodate new data 

998 alter_table_cascade: boolean 

999 Will drop dependent objects when attempting to alter the table. If ``alter_table`` 

1000 is ``False``, this will be ignored. 

1001 from_s3: boolean 

1002 Instead of specifying a table_obj (set the first argument to None), 

1003 set this to True and include :func:`~parsons.databases.Redshift.copy_s3` arguments 

1004 to upsert a pre-existing s3 file into the target_table 

1005 distkey: str 

1006 The column name of the distkey. If not provided, will default to ``primary_key``. 

1007 sortkey: str or list 

1008 The column name(s) of the sortkey. If not provided, will default to ``primary_key``. 

1009 **copy_args: kwargs 

1010 See :func:`~parsons.databases.Redshift.copy` for options. 

1011 """ 

1012 

1013 if isinstance(primary_key, str): 

1014 primary_keys = [primary_key] 

1015 else: 

1016 primary_keys = primary_key 

1017 

1018 # Set distkey and sortkey to argument or primary key. These keys will be used 

1019 # for the staging table and, if it does not already exist, the destination table. 

1020 distkey = distkey or primary_keys[0] 

1021 sortkey = sortkey or primary_key 

1022 

1023 if not self.table_exists(target_table): 

1024 logger.info( 

1025 "Target table does not exist. Copying into newly \ 

1026 created target table." 

1027 ) 

1028 self.copy(table_obj, target_table, distkey=distkey, sortkey=sortkey) 

1029 return None 

1030 

1031 if alter_table and table_obj: 

1032 # Make target table column widths match incoming table, if necessary 

1033 self.alter_varchar_column_widths( 

1034 table_obj, target_table, drop_dependencies=alter_table_cascade 

1035 ) 

1036 

1037 noise = f"{random.randrange(0, 10000):04}"[:4] 

1038 date_stamp = datetime.datetime.now().strftime("%Y%m%d_%H%M") 

1039 # Generate a temp table like "table_tmp_20200210_1230_14212" 

1040 staging_tbl = "{}_stg_{}_{}".format(target_table, date_stamp, noise) 

1041 

1042 if distinct_check: 

1043 primary_keys_statement = ", ".join(primary_keys) 

1044 diff = self.query( 

1045 f""" 

1046 select ( 

1047 select count(*) 

1048 from {target_table} 

1049 ) - ( 

1050 SELECT COUNT(*) from ( 

1051 select distinct {primary_keys_statement} 

1052 from {target_table} 

1053 ) 

1054 ) as total_count 

1055 """ 

1056 ).first 

1057 if diff > 0: 

1058 raise ValueError("Primary key column contains duplicate values.") 

1059 

1060 with self.connection() as connection: 

1061 try: 

1062 # Copy to a staging table 

1063 logger.info(f"Building staging table: {staging_tbl}") 

1064 if "compupdate" not in copy_args: 

1065 # Especially with a lot of columns, compupdate=True can 

1066 # cause a lot of processing/analysis by Redshift before upload. 

1067 # Since this is a temporary table, setting compression for each 

1068 # column is not impactful barely impactful 

1069 # https://docs.aws.amazon.com/redshift/latest/dg/c_Loading_tables_auto_compress.html 

1070 copy_args = dict(copy_args, compupdate=False) 

1071 

1072 if from_s3: 

1073 if table_obj is not None: 

1074 raise ValueError( 

1075 "upsert(... from_s3=True) requires the first argument (table_obj)" 

1076 " to be None. from_s3 and table_obj are mutually exclusive." 

1077 ) 

1078 self.copy_s3(staging_tbl, template_table=target_table, **copy_args) 

1079 else: 

1080 self.copy( 

1081 table_obj, 

1082 staging_tbl, 

1083 template_table=target_table, 

1084 alter_table=False, # We just did our own alter table above 

1085 distkey=distkey, 

1086 sortkey=sortkey, 

1087 **copy_args, 

1088 ) 

1089 

1090 staging_table_name = staging_tbl.split(".")[1] 

1091 target_table_name = target_table.split(".")[1] 

1092 

1093 # Delete rows 

1094 comparisons = [ 

1095 f"{staging_table_name}.{primary_key} = {target_table_name}.{primary_key}" 

1096 for primary_key in primary_keys 

1097 ] 

1098 where_clause = " and ".join(comparisons) 

1099 

1100 sql = f""" 

1101 DELETE FROM {target_table} 

1102 USING {staging_tbl} 

1103 WHERE {where_clause} 

1104 """ 

1105 self.query_with_connection(sql, connection, commit=False) 

1106 logger.debug(f"Target rows deleted from {target_table}.") 

1107 

1108 # Insert rows 

1109 # ALTER TABLE APPEND would be more efficient, but you can't run it in a 

1110 # transaction block. It's worth the performance hit to not commit until the 

1111 # end. 

1112 sql = f""" 

1113 INSERT INTO {target_table} 

1114 SELECT * FROM {staging_tbl}; 

1115 """ 

1116 

1117 self.query_with_connection(sql, connection, commit=False) 

1118 logger.info(f"Target rows inserted to {target_table}") 

1119 

1120 finally: 

1121 if cleanup_temp_table: 

1122 # Drop the staging table 

1123 self.query_with_connection( 

1124 f"DROP TABLE IF EXISTS {staging_tbl};", connection, commit=False 

1125 ) 

1126 logger.info(f"{staging_tbl} staging table dropped.") 

1127 

1128 # Vacuum table. You must commit when running this type of transaction. 

1129 if vacuum: 

1130 with self.connection() as connection: 

1131 connection.set_session(autocommit=True) 

1132 self.query_with_connection(f"VACUUM {target_table};", connection) 

1133 logger.info(f"{target_table} vacuumed.")

1135 def drop_dependencies_for_cols(self, schema, table, cols): 

1136 fmt_cols = ", ".join([f"'{c}'" for c in cols]) 

1137 sql_depend = f""" 

1138 select 

1139 distinct dependent_ns.nspname||'.'||dependent_view.relname as table_name 

1140 from pg_depend 

1141 join pg_rewrite 

1142 on pg_depend.objid = pg_rewrite.oid 

1143 join pg_class as dependent_view 

1144 on pg_rewrite.ev_class = dependent_view.oid 

1145 join pg_class as source_table 

1146 on pg_depend.refobjid = source_table.oid 

1147 join pg_attribute 

1148 on pg_depend.refobjid = pg_attribute.attrelid 

1149 and pg_depend.refobjsubid = pg_attribute.attnum 

1150 join pg_namespace dependent_ns 

1151 on dependent_ns.oid = dependent_view.relnamespace 

1152 join pg_namespace source_ns on source_ns.oid = source_table.relnamespace 

1153 where 

1154 source_ns.nspname = '{schema}' 

1155 and source_table.relname = '{table}' 

1156 and pg_attribute.attname in ({fmt_cols}) 

1157 ; 

1158 """ 

1159 

1160 with self.connection() as connection: 

1161 connection.set_session(autocommit=True) 

1162 tbl = self.query_with_connection(sql_depend, connection) 

1163 dropped_views = [row["table_name"] for row in tbl] 

1164 if dropped_views: 

1165 sql_drop = "\n".join([f"drop view {view} CASCADE;" for view in dropped_views]) 

1166 tbl = self.query_with_connection(sql_drop, connection) 

1167 logger.info(f"Dropped the following views: {dropped_views}") 

1168 

1169 return tbl

1171 def alter_varchar_column_widths(self, tbl, table_name, drop_dependencies=False): 

1172 """ 

1173 Alter the width of a varchar columns in a Redshift table to match the widths 

1174 of a Parsons table. The columns are matched by column name and not their 

1175 index. 

1176 

1177 `Args:` 

1178 tbl: obj 

1179 A Parsons table 

1180 table_name: 

1181 The target table name (e.g. ``my_schema.my_table``) 

1182 `Returns:` 

1183 ``None`` 

1184 """ 

1185 

1186 # Make the Parsons table column names match valid Redshift names 

1187 tbl.table = petl.setheader(tbl.table, self.column_name_validate(tbl.columns)) 

1188 

1189 # Create a list of column names and max width for string values. 

1190 pc = {c: tbl.get_column_max_width(c) for c in tbl.columns} 

1191 

1192 # Determine the max width of the varchar columns in the Redshift table 

1193 s, t = self.split_full_table_name(table_name) 

1194 cols = self.get_columns(s, t) 

1195 rc = {k: v["max_length"] for k, v in cols.items() if v["data_type"] == "character varying"} 

1196 

1197 # Figure out if any of the destination table varchar columns are smaller than the 

1198 # associated Parsons table columns. If they are, then alter column types to expand 

1199 # their width. 

1200 for c in set(rc.keys()).intersection(set(pc.keys())): 

1201 if rc[c] < pc[c] and rc[c] != 65535: 

1202 logger.info(f"{c} not wide enough. Expanding column width.") 

1203 # If requested size is larger than Redshift will allow, 

1204 # automatically set to Redshift's max varchar width 

1205 new_size = 65535 

1206 if pc[c] < new_size: 

1207 new_size = pc[c] 

1208 if drop_dependencies: 

1209 self.drop_dependencies_for_cols(s, t, [c]) 

1210 self.alter_table_column_type(table_name, c, "varchar", varchar_width=new_size)

1213 """ 

1214 Alter a column type of an existing table. 

1215 

1216 table_name: str 

1217 The table name (ex. ``my_schema.my_table``). 

1218 column_name: str 

1219 The target column name 

1220 data_type: str 

1221 A valid Redshift data type to alter the table to. 

1222 varchar_width: 

1223 The new width of the column if of type varchar. 

1224 """ 

1225

1227 

1228 if varchar_width: 

1229 sql += f"({varchar_width})" 

1230 

1231 with self.connection() as connection: 

1232 connection.set_session(autocommit=True) 

1233 self.query_with_connection(sql, connection) 

1234 logger.info(f"Altered {table_name} {column_name}.")
