34class Box(object): 

35 """Box is a file storage provider. 

36 

37 `Args:` 

38 client_id: str 

39 Box client (account) id -- probably a 16-char alphanumeric. 

40 Not required if ``BOX_CLIENT_ID`` env variable is set. 

41 client_secret: str 

42 Box private key -- probably a 32-char alphanumeric. 

43 Not required if ``BOX_CLIENT_SECRET`` env variable is set. 

44 access_token: str 

45 Box developer access token -- probably a 32-char alphanumeric. 

46 Note that this is only valid for developer use only, and should not 

47 be used when creating and maintaining access for typical users. 

48 Not required if ''BOX_ACCESS_TOKEN'' env variable is set. 

49 `Returns:` 

50 Box class 

51 

52 *NOTE*: All path-based methods in this class use an intermediate 

53 method that looks up the relevant folder/file id by successive API 

54 calls. This can be slow for long paths or intermediate folders 

55 that contain many items. If performance is an issue, please use the 

56 corresponding folder_id/file_id methods for each function. 

57 """

165 def list_items_by_id(self, folder_id=DEFAULT_FOLDER_ID, item_type=None) -> Table: 

166 url = "https://api.box.com/2.0/folders/" + folder_id 

167 json_response = self.client.make_request("GET", url) 

168 

169 items = Table(json_response.json()["item_collection"]["entries"]) 

170 if item_type: 

171 items = items.select_rows(lambda row: row.type == item_type) 

172 return items

223 def upload_table_to_folder_id( 

224 self, table, file_name, folder_id=DEFAULT_FOLDER_ID, format="csv" 

225 ) -> boxsdk.object.file.File: 

226 """Save the passed table to Box. 

227 

228 `Args`: 

229 table:Table 

230 The Parsons table to be saved. 

231 file_name: str 

232 The filename under which it should be saved in Box. 

233 folder_id: str 

234 Optionally, the id of the subfolder in which it should be saved. 

235 format: str 

236 For now, only 'csv' and 'json'; format in which to save table. 

237 

238 `Returns`: BoxFile 

239 A Box File object 

240 """ 

241 

242 if format not in self.ALLOWED_FILE_FORMATS: 

243 raise ValueError( 

244 f"Format argument to upload_table() must be in one " 

245 f'of {self.ALLOWED_FILE_FORMATS}; found "{format}"' 

246 ) 

247 

248 # Create a temp directory in which we will let Parsons create a 

249 # file. Both will go away automatically when we leave scope. 

250 with tempfile.TemporaryDirectory() as temp_dir_name: 

251 temp_file_path = temp_dir_name + "/table.tmp" 

252 if format == "csv": 

253 table.to_csv(local_path=temp_file_path) 

254 elif format == "json": 

255 table.to_json(local_path=temp_file_path) 

256 else: 

257 raise SystemError( 

258 f'Got (theoretically) impossible format option "{format}"' 

259 ) # pragma: no cover 

260 

261 new_file = self.client.folder(folder_id).upload( 

262 file_path=temp_file_path, file_name=file_name 

263 ) 

264 return new_file

342 def get_item_id(self, path, base_folder_id=DEFAULT_FOLDER_ID) -> str: 

343 """Given a path-like object, try to return the id for the file or 

344 folder at the end of the path. 

345 

346 *NOTE*: This method makes one API call for each level in 

347 `path`, so can be slow for long paths or intermediate folders 

348 containing very many items. 

349 

350 `Args`: 

351 path: str 

352 A slash-separated path from the base folder to the file or 

353 folder in question. 

354 base_folder_id: str 

355 What to use as the base folder for the path. By default, use 

356 the default folder. 

357 

358 `Returns`: Table 

359 A Parsons Table. 

360 

361 """ 

362 try: 

363 # Grab the leftmost element in the path - this is what we're 

364 # looking for in this folder. 

365 if "/" in path: 

366 this_element, path = path.split(sep="/", maxsplit=1) 

367 if path == "": 

368 raise ValueError('Illegal trailing "/" in file path') 

369 

370 else: 

371 this_element = path 

372 path = "" 

373 

374 # Look in our current base_folder for an item whose name matches the 

375 # current element. If we're at initial, non-recursed call, base_folder 

376 # will be default folder. 

377 item_id = None 

378 for item in self.client.folder(folder_id=base_folder_id).get_items(): 

379 if item.name == this_element: 

380 item_id = item.id 

381 break 

382 

383 if item_id is None: 

384 raise ValueError(f'No file or folder named "{this_element}"') 

385 

386 # If there are no more elements left in path, this is the item we're after. 

387 if not len(path): 

388 return item_id 

389 

390 # If there *are* more elements in the path, we need to check that this item is 

391 # in fact a folder so we can recurse and search inside it. 

392 if item.type != "folder": 

393 raise ValueError(f'Invalid folder "{this_element}"') 

394 

395 return self.get_item_id(path=path, base_folder_id=item_id) 

396 

397 # At the top level of the recursion, when we have the entire path, we want 

398 # to attach it to the error message. If caught at some inner level of the 

399 # recursion, just pass it on up. 

400 except ValueError as e: 

401 if base_folder_id == DEFAULT_FOLDER_ID: 

402 raise ValueError(f'{e}: "{path}"') 

403 else: 

404 raise
